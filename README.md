<article>
  <p>Initialize a Flagstone instance and pass in the required properties</p>
  <code>
  var fs = new FLAGSTONE({
  area: $('.js-floor'),
  flagstones: $('.js-tile'),
  minWidth: 300,
  margin: 10
});
</code>
